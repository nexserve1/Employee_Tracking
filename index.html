4
Create Google Apps Script (Web App)
This acts as your backend API

In your Google Sheet: click Extensions → Apps Script → delete all existing code → paste this:

const SHEET_ID = 'YOUR_SPREADSHEET_ID_HERE'; function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { const params = e.parameter; const postData = e.postData ? JSON.parse(e.postData.contents || '{}') : {}; const action = params.action || postData.action; const sheet_name = params.sheet || postData.sheet; try { const ss = SpreadsheetApp.openById(SHEET_ID); let result; switch(action) { case 'getAll': result = getAll(ss, sheet_name); break; case 'append': result = appendRow(ss, sheet_name, postData.data); break; case 'update': result = updateRow(ss, sheet_name, postData.id, postData.data); break; case 'delete': result = deleteRow(ss, sheet_name, postData.id); break; case 'findOne': result = findOne(ss, sheet_name, params.field, params.value); break; default: result = {error: 'Unknown action'}; } return ContentService .createTextOutput(JSON.stringify({success: true, data: result})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({success: false, error: err.message})) .setMimeType(ContentService.MimeType.JSON); } } function getAll(ss, sheetName) { const sheet = ss.getSheetByName(sheetName); if (!sheet) return []; const values = sheet.getDataRange().getValues(); if (values.length < 2) return []; const headers = values[0]; return values.slice(1).map(row => { const obj = {}; headers.forEach((h, i) => obj[h] = row[i] !== undefined ? String(row[i]) : ''); return obj; }).filter(r => r.id && r.id !== ''); } function appendRow(ss, sheetName, data) { const sheet = ss.getSheetByName(sheetName); if (!sheet) return false; const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; const row = headers.map(h => data[h] !== undefined ? data[h] : ''); sheet.appendRow(row); return true; } function updateRow(ss, sheetName, id, data) { const sheet = ss.getSheetByName(sheetName); if (!sheet) return false; const values = sheet.getDataRange().getValues(); const headers = values[0]; const idCol = headers.indexOf('id'); if (idCol === -1) return false; for (let i = 1; i < values.length; i++) { if (String(values[i][idCol]) === String(id)) { headers.forEach((h, col) => { if (data[h] !== undefined) { sheet.getRange(i + 1, col + 1).setValue(data[h]); } }); return true; } } return false; } function deleteRow(ss, sheetName, id) { const sheet = ss.getSheetByName(sheetName); if (!sheet) return false; const values = sheet.getDataRange().getValues(); const headers = values[0]; const idCol = headers.indexOf('id'); if (idCol === -1) return false; for (let i = 1; i < values.length; i++) { if (String(values[i][idCol]) === String(id)) { sheet.deleteRow(i + 1); return true; } } return false; } function findOne(ss, sheetName, field, value) { const all = getAll(ss, sheetName); return all.find(r => String(r[field]) === String(value)) || null; }

After pasting: Replace YOUR_SPREADSHEET_ID_HERE with your actual Sheet ID from the URL.
Sheet URL looks like: https://docs.google.com/spreadsheets/d/SHEET_ID_IS_HERE/edit

5
Deploy as Web App
Get your API URL
  1. In Apps Script editor: click Deploy → New deployment
  2. Click gear icon ⚙️ next to "Select type" → choose Web app
  3. Set Execute as: Me
  4. Set Who has access: Anyone (anonymous)
  5. Click Deploy
  6. Click Authorize access → choose your Google account → Allow
  7. Copy the Web app URL that appears
⚠️ The URL will look like: https://script.google.com/macros/s/AKfycby.../exec
6
Enter Your Web App URL
Paste the deployed URL to connect

Demo Mode uses localStorage — only works on same device. Use Google Sheets for global access.
